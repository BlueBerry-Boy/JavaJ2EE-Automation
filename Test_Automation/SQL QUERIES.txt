SELECT CONTACT_ID,CUSTOMER_ID, FIRST_NAME,LAST_NAME, HOME_PHONE, EMAIL_ADDRESS FROM CONTACT

SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT

SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
ORDER BY FIRST_NAME DESC 

--COMPARISON OPERATOR--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE CUSTOMER_ID <> 5
ORDER BY FIRST_NAME  

--LIST--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE CUSTOMER_ID <> 5
ORDER BY FIRST_NAME  

SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE FIRST_NAME IN ('ARTHUR', 'WESTLEY')
ORDER BY FIRST_NAME  

--RANGE--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE CUSTOMER_ID BETWEEN 10 AND 20
ORDER BY FIRST_NAME  

--LOGICAL--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE CUSTOMER_ID < 5 AND CONTACT_ID >1000
ORDER BY FIRST_NAME  


--NULL--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS
FROM CONTACT
WHERE CUSTOMER_ID IS NOT NULL
ORDER BY FIRST_NAME  

--PATTERN MATCHING--
SELECT CONTACT_ID, CUSTOMER_ID, FIRST_NAME + LAST_NAME AS NAME , HOME_PHONE, EMAIL_ADDRESS

WHERE FIRST_NAME LIKE 'ADINA%'
ORDER BY FIRST_NAME
